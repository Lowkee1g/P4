FIND_MAX_CROSSING_SUBARRAY (A, low, mid, high)
    left_sum = -∞
    sum = 0
    for i = mid downto low
        sum = sum + A[i]
        if sum > left_sum
            left_sum = sum
            max_left = i
    right_sum = -∞
    sum = 0
    for j = mid + 1 to high 
        sum = sum + A[j]
        if sum > right_sum
            right_sum = sum
            max_right = j
    return (max_left, max_right, left_sum + right_sum)
